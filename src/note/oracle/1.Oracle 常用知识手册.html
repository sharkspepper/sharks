一、基础查询 & 常用函数

伪列：ROWNUM / ROWID — ROWNUM 对结果集编号，ROWID 为物理地址。

-- 取前5行
SELECT ROWNUM, empno, ename FROM emp WHERE ROWNUM <= 5;
-- 利用ROWID快速定位
SELECT * FROM emp WHERE ROWID = 'AAAR3sAAEAAAACXAAA';

字符串函数速查 (|| 拼接, NVL, DECODE)

SELECT 
    'A'||'-'||'B' AS concat_demo,
    NVL(NULL, '默认值') AS nvl_eg,
    DECODE(2,1,'一',2,'二','其他') AS decode_eg,
    SUBSTR('Oracle',2,3) AS sub_demo,   -- 'rac'
    INSTR('HelloWorld','o') AS pos,
    REPLACE('Hello World','World','Oracle') AS rep
FROM DUAL;

日期函数核心

SELECT 
    SYSDATE, 
    SYSDATE + 1 AS 明天,
    MONTHS_BETWEEN(SYSDATE, DATE '2025-01-01') AS 相差月数,
    LAST_DAY(SYSDATE) AS 月末,
    TRUNC(SYSDATE, 'MM') AS 月初,
    TO_CHAR(SYSDATE, 'yyyy-mm-dd hh24:mi:ss') AS 格式化时间
FROM DUAL;
💡 转换三剑客: TO_CHAR, TO_DATE, TO_NUMBER — 灵活处理数据类型转换。

二、高级查询技巧

标准分页 (ROWNUM 三层嵌套)

SELECT * FROM (
    SELECT t.*, ROWNUM rn FROM (
        SELECT * FROM emp ORDER BY empno
    ) t WHERE ROWNUM <= 20
) WHERE rn >= 11;   -- 第2页,每页10条
-- 12c+ 支持 OFFSET FETCH
SELECT * FROM emp ORDER BY empno OFFSET 10 ROWS FETCH NEXT 10 ROWS ONLY;

分析函数 (窗口函数) —— 排名/累计/前后行

SELECT empno, deptno, sal,
    ROW_NUMBER() OVER(PARTITION BY deptno ORDER BY sal DESC) rn,
    RANK()       OVER(PARTITION BY deptno ORDER BY sal DESC) rk,
    SUM(sal) OVER(PARTITION BY deptno) AS dept_total,
    LAG(sal,1,0) OVER(ORDER BY empno) AS prev_sal
FROM emp;

🌲 层次查询 (树形结构) 

SELECT employee_id, manager_id, LEVEL,
       LPAD(' ', 2*(LEVEL-1)) || ename AS tree_name
FROM emp
START WITH manager_id IS NULL
CONNECT BY PRIOR employee_id = manager_id
ORDER SIBLINGS BY ename;

集合运算：UNION / INTERSECT / MINUS

三、DDL / DML & 索引约束

-- 创建表(带注释)
CREATE TABLE employees (
    id NUMBER(10) PRIMARY KEY,
    name VARCHAR2(50) NOT NULL,
    salary NUMBER(10,2) CHECK (salary > 0),
    hire_date DATE DEFAULT SYSDATE
);
COMMENT ON TABLE employees IS '员工表';

-- 索引(普通/唯一/函数索引)
CREATE INDEX idx_emp_dept ON emp(deptno);
CREATE UNIQUE INDEX idx_emp_ename ON emp(ename);
CREATE INDEX idx_emp_upper_name ON emp(UPPER(ename));

-- 约束管理
ALTER TABLE emp ADD CONSTRAINT pk_emp PRIMARY KEY(empno);
ALTER TABLE emp ADD CONSTRAINT fk_dept FOREIGN KEY(deptno) REFERENCES dept(deptno);
-- 禁用/启用
ALTER TABLE emp DISABLE CONSTRAINT fk_dept;
📌 位图索引适用于低基数列，反向键索引减少热块竞争。

四、PL/SQL 编程核心

匿名块 & 异常处理

DECLARE
    v_ename emp.ename%TYPE;
BEGIN
    SELECT ename INTO v_ename FROM emp WHERE empno = 7369;
    DBMS_OUTPUT.PUT_LINE('姓名:'||v_ename);
EXCEPTION
    WHEN NO_DATA_FOUND THEN
        DBMS_OUTPUT.PUT_LINE('无数据');
    WHEN OTHERS THEN
        DBMS_OUTPUT.PUT_LINE('错误:'||SQLERRM);
END;
/

存储过程 + 函数示例

CREATE OR REPLACE PROCEDURE raise_sal(p_empno NUMBER, p_percent NUMBER) AS
BEGIN
    UPDATE emp SET sal = sal * (1 + p_percent/100) WHERE empno = p_empno;
    COMMIT;
END;
/

CREATE OR REPLACE FUNCTION get_annual(p_empno NUMBER) RETURN NUMBER IS
    v_sal emp.sal%TYPE;
BEGIN
    SELECT sal INTO v_sal FROM emp WHERE empno = p_empno;
    RETURN v_sal * 12;
END;
/

游标 (显式 & FOR循环)

BEGIN
    FOR rec IN (SELECT ename, sal FROM emp WHERE deptno=20) LOOP
        DBMS_OUTPUT.PUT_LINE(rec.ename||' - '||rec.sal);
    END LOOP;
END;
/

五、性能优化 & 执行计划

查看执行计划三种方式

-- 1. EXPLAIN PLAN
EXPLAIN PLAN FOR SELECT * FROM emp WHERE deptno=10;
SELECT * FROM TABLE(DBMS_XPLAN.DISPLAY);

-- 2. AUTOTRACE (SQL*Plus)
SET AUTOTRACE TRACEONLY;
SELECT * FROM emp;

-- 3. 查看真实游标执行计划
SELECT * FROM TABLE(DBMS_XPLAN.DISPLAY_CURSOR('sql_id', NULL, 'ALLSTATS LAST'));

常用优化技巧

  • ✔ 避免在索引列上使用函数：UPPER(name)= 'SMITH' → 可改用函数索引
  • ✔ 用 EXISTS 代替 IN (当子查询返回大量数据)
  • ✔ 合理使用 /*+ INDEX(table_name idx_name) */ 提示
  • ✔ 批量 DML 使用 FORALL 减少上下文切换
📊 统计信息过时会导致执行计划错误，定期收集: DBMS_STATS.GATHER_TABLE_STATS('SCOTT','EMP');

📌 用户视图

USER_TAB_COLUMNS (列信息)
📌 动态性能视图
V$SESSION · V$SQL · V$LOCK · V$PROCESS
用于实时监控会话、锁等待和SQL统计

闪回查询 (挽回误操作)

-- 查询15分钟前数据
SELECT * FROM emp AS OF TIMESTAMP (SYSTIMESTAMP - INTERVAL '15' MINUTE);

-- 闪回表到特定时间点 (需启用行移动)
ALTER TABLE emp ENABLE ROW MOVEMENT;
FLASHBACK TABLE emp TO TIMESTAMP (SYSTIMESTAMP - INTERVAL '30' MINUTE);

序列 / 同义词 / 物化视图

CREATE SEQUENCE seq_emp_id START WITH 100 INCREMENT BY 1;
SELECT seq_emp_id.NEXTVAL FROM DUAL;
CREATE SYNONYM my_emp FOR scott.emp;
CREATE MATERIALIZED VIEW mv_sum_sal REFRESH FAST ON COMMIT AS SELECT deptno,SUM(sal) FROM emp GROUP BY deptno;

🔁 事务控制: COMMIT, ROLLBACK, SAVEPOINT sp1; ROLLBACK TO sp1;
-- 示例: 设置保存点
UPDATE accounts SET balance = balance - 100 WHERE id=1;
SAVEPOINT sp_deduct;
UPDATE accounts SET balance = balance + 100 WHERE id=2;
-- 若发现错误回滚到保存点
ROLLBACK TO sp_deduct;
COMMIT;

九、日常运维 & 诊断片段

查看当前会话锁等待

SELECT a.sid, a.serial#, b.id1, c.sql_text 
FROM v$session a, v$lock b, v$sql c 
WHERE a.sid = b.sid AND a.sql_id = c.sql_id AND b.block > 0;

查看最消耗资源的SQL (过去一小时)

SELECT sql_id, cpu_time, elapsed_time, buffer_gets, sql_text
FROM v$sql
WHERE last_active_time > SYSDATE - 1/24
ORDER BY cpu_time DESC
FETCH FIRST 10 ROWS ONLY;

表空间使用率

SELECT tablespace_name, ROUND(SUM(bytes)/1024/1024,2) MB,
       ROUND(SUM(maxbytes)/1024/1024,2) max_mb
FROM dba_data_files GROUP BY tablespace_name;
🔐 生产环境慎用自动undo retention调整，避免ORA-01555快照过旧。

十、快速记忆 & 最佳实践

  • 分页模板：ROWNUM 三层 + 排序子查询，12c后推荐OFFSET FETCH。
  • 避免隐式转换：类型不匹配导致索引失效。
  • 批量DML：FORALL + BULK COLLECT 提升性能。
  • 绑定变量：减少硬解析，提高共享池命中率。
  • 执行计划稳定：SQL Profile 或 SQL Plan Baseline。
🧠 Oracle 内核精华: 一致性读、回滚段、多版本并发控制(CR) 是理解性能问题的基石。
